301 void leveleditor_eventhandler()
302 {
303  while ( SDL_PollEvent( &event ) )
304  {
305  // First Let's poll for Mouse button HELD Down
306  // You can hold right click to delete tiles constantly
307  Uint8 ms = SDL_GetMouseState(NULL,NULL);
308  if (ms & SDL_BUTTON(SDL_BUTTON_RIGHT))
309  {
310  pLevelEditor->DeleteObject();
311  }
312 
313  switch ( event.type )
314  {
315  case SDL_QUIT:
316  {
317  done = 2;
318  break;
319  }
320  case SDL_KEYDOWN:
321  {
322  // ESCAPE to ESCAPE mouse command or LEVEL MODE
323  if( event.key.keysym.sym == SDLK_ESCAPE )
324  {
325  // universal!
326  if (pLevelEditor->Mouse_command == MOUSE_COMMAND_NOTHING)
327  mode = MODE_GAME;
328  else
329  pLevelEditor->Mouse_command = MOUSE_COMMAND_NOTHING;
330  }
331  // ~ to enter CONSOLE
332  else if ( event.key.keysym.sym == SDLK_BACKQUOTE )
333  {
334  oldmode = mode;
335  mode = MODE_CONSOLE;
336 
337  }
338  // F8 to Exit LEVLE MODE
339  else if( event.key.keysym.sym == SDLK_F8 )
340  {
341 
342  mode = MODE_GAME;
343 
344  pCamera->SetPos( pPlayer->posx - pCamera->x - Screen->w, 0 );
345 
346  }
347  // IF LCTRL IS HELD
348  else if (event.key.keysym.mod & KMOD_LCTRL)
349  {
350  // LCTRL S to save
351  if( event.key.keysym.sym == SDLK_s )
352  {
353  pLevel->Save();
354  }
355  // LCTRL+C to copy
356  else if( event.key.keysym.sym == SDLK_c )
357  {
358  pLevelEditor->SetCopyObject();
359  }
360  //LCTRL+V to paste
361  else if( event.key.keysym.sym == SDLK_v )
362  {
363  pLevelEditor->PasteObject();
364  }
365  }
366  // F Key for Fast Copy
367  else if (event.key.keysym.sym == SDLK_f)

369  if (pLevelEditor->Mouse_command == MOUSE_COMMAND_FASTCOPY)
370  pLevelEditor->Mouse_command = MOUSE_COMMAND_NOTHING;
371  else

373  }
374  else if( ( event.key.keysym.sym == SDLK_w || event.key.keysym.sym == SDLK_a ||
375  event.key.keysym.sym == SDLK_s || event.key.keysym.sym == SDLK_d ) )
376  {
377  // level editor mode
378  if( pLevelEditor->Mouse_command == MOUSE_COMMAND_FASTCOPY && pLevelEditor->CopyObject)
379  {

381  {

389  }
390  else if( event.key.keysym.sym == SDLK_a )
391  {
392  pLevelEditor->PasteObject( pLevelEditor->CopyObject->posx - pCamera->x - pLevelEditor->CopyObject->width, pLevelEditor->CopyObject->posy - pCamera->y );
393  pCamera->Move( - pLevelEditor->CopyObject->width, 0 );
394  }
395  else if( event.key.keysym.sym == SDLK_s )
396  {
397  pLevelEditor->PasteObject( pLevelEditor->CopyObject->posx- pCamera->x , pLevelEditor->CopyObject->posy - pCamera->y + pLevelEditor->CopyObject->height );
398  pCamera->Move( 0, pLevelEditor->CopyObject->height );
399  }
400  }
401  else
402  {
403  // level editor mode
404  if( event.key.keysym.sym == SDLK_d)
405  {
